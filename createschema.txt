CREATE TABLE BOOK(
   isbn      VARCHAR(13) NOT NULL PRIMARY KEY
  ,title     VARCHAR(200)
  ,author    VARCHAR(200)
  ,price     VARCHAR(6)
  ,publisher VARCHAR(200)
  ,imagesrc  VARCHAR(200)
  ,genre     INTEGER 
);

CREATE TABLE USERS (
	uid INT PRIMARY KEY AUTO_INCREMENT,
	admin BOOLEAN NOT NULL,
	uname VARCHAR(40),
	upasshashed VARCHAR(40),
	salt VARCHAR(40),
	fname VARCHAR(40),
	lname VARCHAR(40)
);
CREATE TABLE ORDERS (
	oid INT PRIMARY KEY AUTO_INCREMENT,
	uid INT,
	isbn VARCHAR(13)
);
CREATE TABLE CARTS (
	cid INT PRIMARY KEY AUTO_INCREMENT,
	uid INT,
	isbn VARCHAR(13)
); 
ALTER TABLE ORDERS ADD CONSTRAINT FK1 FOREIGN KEY (uid) REFERENCES USERS(uid) ON DELETE CASCADE ON UPDATE RESTRICT;
ALTER TABLE ORDERS ADD CONSTRAINT FK2 FOREIGN KEY (isbn) REFERENCES BOOK(isbn) ON DELETE CASCADE ON UPDATE RESTRICT;
ALTER TABLE CARTS ADD CONSTRAINT FK3 FOREIGN KEY (uid) REFERENCES USERS(uid) ON DELETE CASCADE ON UPDATE RESTRICT;
ALTER TABLE CARTS ADD CONSTRAINT FK4 FOREIGN KEY (isbn) REFERENCES BOOK(isbn) ON DELETE CASCADE ON UPDATE RESTRICT;
